\section{Notation}
First order differential of y w.r.t x
\begin{equation}
	\frac{d}{dx} y
\end{equation}
\begin{equation}
\dot y
\end{equation}
Second order differential of y w.r.t x
\begin{equation}
	\frac{d^2}{dx^2} y
\end{equation}
\begin{equation}
	\ddot y
\end{equation}


\chapter{Differentiation}

The derivative of a function is defined as

\begin{equation}
	\frac{df}{dx} = \lim\limits_{h - 0} \frac{f (x + h) - f (x)}{h}
\end{equation}

\section{Chain Rule}

Given

\begin{equation}
	f(x) = F(g(x))
\end{equation}
Then
\begin{equation}
	\frac{d}{dx} f = \frac{d}{dg} F \cdot \frac{d}{dx} g
\end{equation}

\section{Product Rule}

Given

\begin{equation}
	f(x) = u(x)v(x)
\end{equation}
Then
\begin{equation}
	\frac{d}{dx} f= \frac{d}{dx} u\cdot v + u \cdot \frac{d}{dx} v
\end{equation}
