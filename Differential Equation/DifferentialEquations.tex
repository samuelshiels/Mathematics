	\frac{dy}{dx}

\begin{equation}
	\frac{d^2y}{dx^2}
\end{equation}

\chapter{Differentiation}

The derivative of a function is defined as

\begin{equation}
	\frac{df}{dx} = \lim\limits_{h - 0} \frac{f (x + h) - f (x)}{h}
\end{equation}

\section{Chain Rule}

Given

\begin{equation}
	f(x) = F(g(x))
\end{equation}
Then
\begin{equation}
	\frac{df}{dx} = \frac{dF}{dg} \cdot \frac{dg}{dx}
\end{equation}

\section{Product Rule}

Given

\begin{equation}
	f(x) = u(x)v(x)
\end{equation}
Then
\begin{equation}
	\frac{df}{dx} = \frac{du}{dx}(x)\cdot v(x) + u(x) \cdot \frac{dv}{dx}
\end{equation}
